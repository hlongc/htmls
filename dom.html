<button id="app" onclick="fn1();fn2()">dom0级事件</button>
    <script>
      function fn1() {
        console.log('fn1')
      }
      function fn2() {
        console.log('fn2')
      }
      app.onclick = function() {
        console.log('111')
      }
      app.onclick = function() {
        console.log('222')
      }
      // print 222
    </script>
    
      <button id="root">dom2级事件</button>
      <script>
        function handleClick() {
          console.log('dom2级：addEventListener')
        }
        root.addEventListener('click', handleClick)
        root.addEventListener('click', handleClick)
        root.addEventListener("click", function() {
          console.log('匿名函数')
        })
        // dom2级：addEventListener
        // 匿名函数
      </script>
